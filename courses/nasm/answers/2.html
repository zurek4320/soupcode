Question: Write a NASM program displaying your name.

Example Solution 1:

section .data
    message: db "Bob", 10

section .text
    global _start

_start:
    mov eax, 4
    mov ebx, 1
    mov ecx, message
    mov edx, 4 ; Bob is 3 chars and newline (10) is 1 char
    int 80h

    mov eax, 1
    xor ebx, ebx
    int 80h

Example Solution 2:

section .data
    message: db "John", 10
    message_len: db $-message

section .text
    global _start

_start:
    mov eax, 4
    mov ebx, 1
    mov ecx, message
    mov edx, message_len
    int 80h

    mov eax, 1
    xor ebx, ebx
    int 80h
