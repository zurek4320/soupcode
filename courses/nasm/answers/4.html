Question: Write a "What's your name?" NASM program.

Solution:

section .bss
    name resb 64 ; 64 characters space for the name
    name_len resd 1 ; 4 bytes for storing lenght

section .data
    what_name: db "What's your name: "
    what_name_len: equ $-what_name
    hello: db "Hello, "
    hello_len: equ $-hello
    mark: db "!", 10

section .text
    global _start

_start:
    mov eax, 4
    mov ebx, 1
    mov ecx, what_name
    mov edx, what_name_len
    int 80h

    mov eax, 3
    mov ebx, 0
    mov ecx, name
    mov edx, 64
    int 80h
    mov [name_len], eax

    mov eax, 4
    mov ebx, 1
    mov ecx, hello
    mov edx, hello_len
    int 80h

    ; "deleting" newline character from input, you can't just do "sub [name_len], 1"
    mov eax, [name_len]
    sub eax, 1
    mov [name_len], eax

    mov eax, 4
    mov ebx, 1
    mov ecx, name
    mov edx, [name_len]
    int 80h

    mov eax, 4
    mov ebx, 1
    mov ecx, mark
    mov edx, 2
    int 80h

    mov eax, 1
    xor ebx, ebx
    int 80h
