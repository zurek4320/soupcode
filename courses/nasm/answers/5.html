Question: Write a NASM program that counts down from 10 to 1 and then exits.

Solution:

section .text
  global _start

_start:
  mov ecx, 5        ; loop counter
  loop_start:
    ; do something here
    dec ecx
    cmp ecx, 0
    jne loop_start    ; repeat until ecx == 0

    ; exit program
    mov eax, 1
    xor ebx, ebx
    int 80h
